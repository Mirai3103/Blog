import React from 'react';const withAuthentication = (WrappedComponent) => { return class extends React.Component {  constructor(props) {   super(props);   this.state = {    isAuthenticated: false,   };  }  authenticate = () => {   // Logic xác thực người dùng   this.setState({ isAuthenticated: true });  }  render() {   return (    <WrappedComponent     {...this.props}     isAuthenticated={this.state.isAuthenticated}     authenticate={this.authenticate}    />   );  } };};const Button = (props) => { return (  <button onClick={props.authenticate} disabled={!props.isAuthenticated}>   Nhấn nút  </button> );};const ButtonWithAuthentication = withAuthentication(Button);export default ButtonWithAuthentication;
